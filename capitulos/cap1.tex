\begin{defn}

Sea $\mathcal{V}$ un subespacio de funciones definidas de $\mathds{R}^d$ a $A \subseteqq \mathds{R}^d$. Se dice que:
\begin{itemize}
	\item $\mathcal{V}$ es invariante si para  $f \in \mathcal{V}$ y  $A\in\mathcal{O}^d$, entonces  $f_A \in \mathcal{V}$.
	\item $\mathcal{V}$ es reducible si  $\mathcal{V} = \mathcal{V}_1 + \mathcal{V}_2$ con $\mathcal{V}_1 \not= \emptyset$, $\mathcal{V}_2 \not= \emptyset$ verificando $\mathcal{V}_1,\mathcal{V}_2$ irreducibles y $\mathcal{V}_1 \perp \mathcal{V}_2$.
	\item $\mathcal{V}$ es irreducible si no es reducible.
	\item $\mathcal{V}$ es primitivo si es invariante e irreducible.
\end{itemize}
\end{defn}
\begin{defn}Dado $f:\mathds{R}^d \to \mathds{C}$ se define $span(f_A)$ como el espacio de las series convergentes $\sum c_jf_{A_j}$ con $A_j \in \mathcal{O}^d$,$c_j \in \mathds{C}$ 
\end{defn}

Si $\mathcal{V}$ es un espacio finito dimensional $\mathcal{V} = span\{f_A\}$
